Syntax   -> Expr                                         ->   ( id -         ->   #dollar
Expr     -> Term Expr#                                   ->   ( id -         ->   #dollar )
Expr#    -> + Term Expr# | - Term Expr# | #epsilon       ->   + - #epsilon   ->   #dollar )
Term     -> Factor Term#                                 ->   ( id -         ->   #dollar + - )
Term#    -> * Factor Term# | / Factor Term# | #epsilon   ->   * / #epsilon   ->   #dollar + - )
Factor   -> ( Expr ) | id | - Factor                     ->   ( id -         ->   #dollar * / + - )
